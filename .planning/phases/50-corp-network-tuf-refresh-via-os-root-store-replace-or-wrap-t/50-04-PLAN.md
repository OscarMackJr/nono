<!-- The StaticMapTransport shape from RESEARCH.md Example B -->

```rust
use async_trait::async_trait;
use bytes::Bytes;
use futures::stream;
use std::collections::HashMap;
use std::sync::Arc;
use tough::{Transport, TransportError, TransportErrorKind, TransportStream};
use url::Url;

#[derive(Debug, Clone)]
struct StaticMapTransport {
    files: Arc<HashMap<String, Vec<u8>>>,
}

#[async_trait]
impl Transport for StaticMapTransport {
    async fn fetch(&self, url: Url) -> std::result::Result<TransportStream, TransportError> {
        let key = url.path().trim_start_matches('/').to_string();
        match self.files.get(&key) {
            Some(bytes) => {
                let s = stream::iter(std::iter::once(
                    Ok::<Bytes, TransportError>(Bytes::from(bytes.clone())),
                ));
                Ok(Box::pin(s))
            }
            None => Err(TransportError::new(TransportErrorKind::FileNotFound, url)),
        }
    }
}
```

```rust
// Inner helper — takes an injectable Transport so tests can swap it.
// Crate-visible because the test module needs to call it from a sibling mod.
pub(crate) async fn refresh_trusted_root_with_transport(
    transport: impl Transport + Send + Sync + 'static,
    metadata_url: Url,
    targets_url: Url,
    datastore_dir: std::path::PathBuf,
    embedded_root: &[u8],
) -> Result<TrustedRoot> {
    tokio::fs::create_dir_all(&datastore_dir).await.map_err(|e| {
        NonoError::Setup(format!(
            "create tuf-cache dir {}: {e}",
            datastore_dir.display()
        ))
    })?;

    let datastore_for_cleanup = datastore_dir.clone();
    let repo = RepositoryLoader::new(&embedded_root, metadata_url, targets_url)
        .transport(transport)
        .datastore(datastore_dir)
        .load()
        .await
        .map_err(|e| {
            let _ = std::fs::remove_dir_all(&datastore_for_cleanup);
            NonoError::Setup(format!("Sigstore TUF refresh failed: {e}"))
        })?;

    let target_name = TargetName::new(TRUSTED_ROOT_TARGET)
        .map_err(|e| NonoError::Setup(format!("invalid target name: {e}")))?;
    let stream = repo
        .read_target(&target_name)
        .await
        .map_err(|e| NonoError::Setup(format!("read trusted_root target: {e}")))?
        .ok_or_else(|| {
            NonoError::Setup(format!(
                "Sigstore target not found in TUF repo: {TRUSTED_ROOT_TARGET}"
            ))
        })?;

    use tough::IntoVec;
    let bytes = stream
        .into_vec()
        .await
        .map_err(|e| NonoError::Setup(format!("collect trusted_root bytes: {e}")))?;

    let json = std::str::from_utf8(&bytes)
        .map_err(|e| NonoError::Setup(format!("trusted_root.json is not UTF-8: {e}")))?;
    TrustedRoot::from_json(json)
        .map_err(|e| NonoError::Setup(format!("parse trusted_root.json: {e}")))
}

// Public production entrypoint — wraps the inner helper with prod values.
pub async fn refresh_production_trusted_root() -> Result<TrustedRoot> {
    let base_url = Url::parse(DEFAULT_TUF_URL)
        .map_err(|e| NonoError::Setup(format!("invalid Sigstore TUF URL: {e}")))?;
    let metadata_url = base_url.clone();
    let targets_url = base_url
        .join("targets/")
        .map_err(|e| NonoError::Setup(format!("invalid Sigstore targets URL: {e}")))?;
    let datastore_dir = crate::config::nono_home_dir()
        .map_err(|e| NonoError::Setup(format!("resolve nono home dir: {e}")))?
        .join(".nono")
        .join("trust-root")
        .join("tuf-cache");
    let agent = build_corp_friendly_agent();
    let transport = UreqTransport { agent };
    refresh_trusted_root_with_transport(
        transport,
        metadata_url,
        targets_url,
        datastore_dir,
        PRODUCTION_TUF_ROOT,
    )
    .await
}
```

mod tests {
    use super::*;
    use std::collections::HashMap;
    use std::path::PathBuf;
    use std::sync::Arc;
    use tempfile::tempdir;

    /// In-memory tough::Transport for hermetic testing.
    /// D-50-08: no localhost HTTP server, no port allocation, no Defender interference.
    #[derive(Debug, Clone)]
    struct StaticMapTransport {
        files: Arc<HashMap<String, Vec<u8>>>,
    }

    #[async_trait]
    impl Transport for StaticMapTransport {
        async fn fetch(
            &self,
            url: Url,
        ) -> std::result::Result<TransportStream, TransportError> {
            // Key by URL path (strip leading slash so the HashMap layout matches
            // the on-disk fixture filename layout).
            let key = url.path().trim_start_matches('/').to_string();
            match self.files.get(&key) {
                Some(bytes) => {
                    let s = stream::iter(std::iter::once(
                        Ok::<Bytes, TransportError>(Bytes::from(bytes.clone())),
                    ));
                    Ok(Box::pin(s))
                }
                None => Err(TransportError::new(TransportErrorKind::FileNotFound, url)),
            }
        }
    }

    fn fixture_dir(name: &str) -> PathBuf {
        PathBuf::from(env!("CARGO_MANIFEST_DIR"))
            .join("tests")
            .join("fixtures")
            .join(name)
    }

    /// Recursively load every file under `fixture_dir(name)` into a HashMap
    /// keyed by the path relative to the fixture root (using forward slashes
    /// so URL-path keys match on Windows too).
    fn load_fixture(name: &str) -> HashMap<String, Vec<u8>> {
        let root = fixture_dir(name);
        let mut map = HashMap::new();
        fn walk(root: &std::path::Path, prefix: &str, map: &mut HashMap<String, Vec<u8>>) {
            for entry in std::fs::read_dir(root).expect("read fixture dir") {
                let entry = entry.expect("dir entry");
                let path = entry.path();
                let name = entry.file_name().into_string().expect("utf-8 filename");
                let key = if prefix.is_empty() {
                    name.clone()
                } else {
                    format!("{prefix}/{name}")
                };
                if path.is_dir() {
                    walk(&path, &key, map);
                } else {
                    let bytes = std::fs::read(&path).expect("read fixture file");
                    map.insert(key, bytes);
                }
            }
        }
        walk(&root, "", &mut map);
        map
    }

    fn embedded_root_for_fixture(files: &HashMap<String, Vec<u8>>) -> Vec<u8> {
        // The "embedded" anchor for the test is the 1.root.json that ships
        // with the fixture (NOT the production PRODUCTION_TUF_ROOT const,
        // which is signed by sigstore's real keys).
        files
            .get("1.root.json")
            .cloned()
            .expect("1.root.json missing from fixture")
    }

    fn test_urls() -> (Url, Url) {
        (
            Url::parse("http://hermetic.test/").expect("parse metadata url"),
            Url::parse("http://hermetic.test/targets/").expect("parse targets url"),
        )
    }

    /// Covers SPEC Req 5 acceptance bullet (a).

            files: Arc::new(files),

            meta,
            targets,
            datastore.path().join("tuf-cache"),
            &embedded,
        )
        .await;

        assert!(!json.is_empty(), "serialized trusted_root must be non-empty");
        assert!(json.contains('{'), "serialized output must be JSON");
    }

    /// Test 2 — bad signature: tough's chain walk MUST reject and surface
    /// as NonoError::Setup. Covers SPEC Req 3 + SPEC Req 5 bullet (b).
    #[tokio::test]
    async fn bad_signature_at_v15_surfaces_as_nono_error_setup() {

        let embedded = embedded_root_for_fixture(&files);
        let transport = StaticMapTransport {
            files: Arc::new(files),
        };

        let (meta, targets) = test_urls();

            transport,
            meta,
            targets,
            datastore.path().join("tuf-cache"),
            &embedded,
        )
        .await;

        match result {
            Err(NonoError::Setup(msg)) => {
                assert!(
                    msg.contains("Sigstore TUF refresh failed"),
                    "bad-sig error must surface through the TUF-refresh-failed wrapping; got: {msg}"
                );
            }
            Err(other) => panic!("expected NonoError::Setup for bad sig, got {other:?}"),
            Ok(_) => panic!("bad-sig fixture must NOT pass tough's signature verification"),
        }
    }

    /// Test 3 — malformed JSON: surface as NonoError::Setup.
    /// Covers SPEC Req 5 bullet (c).
    #[tokio::test]
    async fn malformed_json_at_v15_surfaces_as_nono_error_setup() {
        let files = load_fixture("tuf-repo-malformed");
        let embedded = embedded_root_for_fixture(&files);
        let transport = StaticMapTransport {
            files: Arc::new(files),
        };

        )
        .await;

        match result {
            Err(NonoError::Setup(_)) => { /* expected */ }
            Err(other) => panic!("expected NonoError::Setup for malformed JSON, got {other:?}"),
            Ok(_) => panic!("malformed-JSON fixture must NOT parse successfully"),
        }
    }

    /// Test 4 — byte-identical cache snapshot. Confirms the Req 4 contract:
    /// the bytes we write to <nono_home>/.nono/trust-root/trusted_root.json
    /// equal `serde_json::to_string_pretty(&TrustedRoot::from_json(<source>)).unwrap()`
    /// for the trusted_root.json target in the happy fixture.
    ///
    /// This is the round-trip idempotency check from RESEARCH.md Pitfall 5:
    /// we compare against the fixture's PARSED-AND-RESERIALIZED bytes, NOT
    /// against the raw fixture bytes (which may differ in key ordering /
    /// whitespace from what serde_json::to_string_pretty produces).
    #[tokio::test]
    async fn cache_bytes_match_baseline() {
        let files = load_fixture("tuf-repo-happy");
        let embedded = embedded_root_for_fixture(&files);

        // depending on consistent_snapshots).
        let baseline_bytes: &[u8] = files
            .iter()
            .find(|(k, _)| k.ends_with("trusted_root.json") && k.starts_with("targets/"))
            .map(|(_, v)| v.as_slice())
            .expect("fixture must contain a targets/.../trusted_root.json file");
        let baseline_json_str =
            std::str::from_utf8(baseline_bytes).expect("baseline bytes are UTF-8");
        let baseline_parsed =
            TrustedRoot::from_json(baseline_json_str).expect("baseline parses");
        let baseline_serialized =
            serde_json::to_string_pretty(&baseline_parsed).expect("baseline reserializes");

        let transport = StaticMapTransport {
            files: Arc::new(files),
        };
        let datastore = tempdir().expect("tempdir for datastore");
        let (meta, targets) = test_urls();
        let trusted_root = refresh_trusted_root_with_transport(
            transport,
            meta,
            targets,
            datastore.path().join("tuf-cache"),
            &embedded,
        )
        .await
        .expect("happy-path returns Ok");

        let actual_serialized =
            serde_json::to_string_pretty(&trusted_root).expect("serialize result");

        assert_eq!(
            actual_serialized.as_bytes(),
            baseline_serialized.as_bytes(),
            "chain-walk output bytes must equal the parsed-and-reserialized baseline bytes (Req 4 byte-identical cache contract)"
        );
    }
}
