    fn nono_trust_root_cache_path() -> Result<PathBuf> {
        // Honor NONO_TEST_HOME (Phase 27.1 D-27.1-08 cross-cut convention) FIRST.
        if let Ok(test_home) = std::env::var("NONO_TEST_HOME") {
            let p = PathBuf::from(&test_home);
            if !p.is_absolute() {
                return Err(NonoError::TrustPolicy(format!(
                    "NONO_TEST_HOME must be an absolute path, got: {test_home}"
                )));
            }
            return Ok(p.join(".nono").join("trust-root").join("trusted_root.json"));
        }
        // Fallback: resolve user home via std::env (mirrors dirs::home_dir()
        // logic but with NO new dep). Per P32-CHK-002 / D-32-15.
        let home = home_dir_from_env().ok_or_else(|| NonoError::TrustPolicy(
            "could not resolve user home directory (set NONO_TEST_HOME, USERPROFILE, or HOME)".to_string()
        ))?;
        Ok(home.join(".nono").join("trust-root").join("trusted_root.json"))
    }

    /// Resolve the user home directory using `std::env` only (no `dirs` dep).
    ///
    /// Mirrors the resolution order of `dirs::home_dir()`:
    /// - Windows: `USERPROFILE`, falling back to `HOMEDRIVE` + `HOMEPATH`.
    /// - Unix:    `HOME`.
    ///
    /// Honors P32-CHK-002 / D-32-15: `crates/nono` MUST NOT add `dirs` as a
    /// production dependency. NONO_TEST_HOME is checked by the caller BEFORE
    /// this function runs (Phase 27.1 D-27.1-08 convention).
    fn home_dir_from_env() -> Option<PathBuf> {
        #[cfg(windows)]
        {
            if let Some(profile) = std::env::var_os("USERPROFILE") {
                let p = PathBuf::from(&profile);
                if !p.as_os_str().is_empty() {
                    return Some(p);
                }
            }
            if let (Some(drive), Some(path)) =
                (std::env::var_os("HOMEDRIVE"), std::env::var_os("HOMEPATH"))
            {
                let mut p = PathBuf::from(drive);
                p.push(path);
                if !p.as_os_str().is_empty() {
                    return Some(p);
                }
            }
            None
        }
        #[cfg(not(windows))]
        {
            std::env::var_os("HOME")
                .map(PathBuf::from)
                .filter(|p| !p.as_os_str().is_empty())
        }
    }

    ```rust
    /// Load the cached Sigstore trusted root from disk.
    ///
    /// Reads `<nono_home_dir()>/.nono/trust-root/trusted_root.json` synchronously.
    /// The cache is hydrated by `nono setup --refresh-trust-root` (which fetches
    /// fresh TUF metadata from `https://tuf-repo-cdn.sigstore.dev`).
    ///
    /// **Diverges from upstream** `sigstore_verify::TrustedRoot::production()` per
    /// Phase 32 D-32-01: this fork caches the trusted root explicitly under a
    /// per-user path to keep the verify path offline. See
    /// `.planning/phases/32-sigstore-integration/32-CONTEXT.md` (D-32-01 / D-32-15).
    /// The upstream-drift sentinel at `tests/integration/test_upstream_drift.sh:257`
    /// is annotated `# intentional fork: Phase 32 D-32-01`.
    ///
    /// # Errors
    /// * `NonoError::TrustPolicy` — when the cache file does not exist (D-32-05
    ///   first-run UX). Recovery: `nono setup --refresh-trust-root`.
    /// * `NonoError::TrustVerification` — when the cache exists but is past its
    ///   `valid_for.end` (D-32-03 expiry gate). Recovery: same.
    pub fn load_production_trusted_root() -> Result<TrustedRoot> {
        let cache_path = nono_trust_root_cache_path()?;

        if !cache_path.exists() {
            return Err(NonoError::TrustPolicy(
                "Sigstore trusted root not initialized; run \
                 `nono setup --refresh-trust-root` (requires network)."
                    .to_string(),
            ));
        }

        let trusted_root = TrustedRoot::from_file(&cache_path)
            .map_err(|e| NonoError::TrustPolicy(format!(
                "failed to load cached trusted root from {}: {e}",
                cache_path.display()
            )))?;

        check_trusted_root_freshness(&trusted_root, &cache_path)?;
        Ok(trusted_root)
    }

    /// Returns Err(TrustVerification) when no active tlog key has `valid_for.end > now`.
    /// At least one tlog must remain valid for keyless verify to succeed.
    ///
    /// Per RESEARCH Pitfall 3: an expired RETIRED tlog is normal (kept for
    /// historical bundle verification); only the absence of any current key
    /// indicates a stale cache.
    fn check_trusted_root_freshness(root: &TrustedRoot, cache_path: &Path) -> Result<()> {
        let now_iso10 = current_date_iso_prefix(); // "YYYY-MM-DD" — see helper below
        let any_active = root.tlogs.iter().any(|tlog| {
            tlog.public_key
                .as_ref()
                .and_then(|pk| pk.valid_for.as_ref())
                .and_then(|v| v.end.as_ref())
                .map(|end| now_iso10.as_str() < &end[..end.len().min(10)])
                .unwrap_or(true) // missing end = no expiry asserted; treat as active
        });
        if any_active {
            return Ok(());
        }
        let latest_end = root.tlogs.iter()
            .filter_map(|tlog| tlog.public_key.as_ref())
            .filter_map(|pk| pk.valid_for.as_ref())
            .filter_map(|v| v.end.as_ref())
            .max()
            .cloned()
            .unwrap_or_else(|| "unknown".to_string());
        Err(NonoError::TrustVerification {
            path: cache_path.display().to_string(),
            reason: format!(
                "Sigstore trusted root expired {latest_end}; run \
                 `nono setup --refresh-trust-root` (requires network)."
            ),
        })
    }

    /// Returns the current UTC date as an ISO-8601 prefix `YYYY-MM-DD`.
    ///
    /// Avoids `chrono` to preserve D-19. Uses std::time::SystemTime + manual
    /// Gregorian conversion. Lossy at sub-day granularity but fail-closed-correct
    /// for D-32-03's "is the cache past its expiry date" question.
    fn current_date_iso_prefix() -> String {
        // Get seconds since UNIX epoch.
        let now_secs = std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH)
            .map(|d| d.as_secs())
            .unwrap_or(0);
        // Days since epoch, then convert to civil date via the Howard Hinnant
        // algorithm (https://howardhinnant.github.io/date_algorithms.html).
        let days = (now_secs / 86_400) as i64;
        let z = days + 719_468;
        let era = if z >= 0 { z / 146_097 } else { (z - 146_096) / 146_097 };
        let doe: u64 = (z - era * 146_097) as u64;
        let yoe: u64 = (doe - doe / 1_460 + doe / 36_524 - doe / 146_096) / 365;
        let y: i64 = (yoe as i64) + era * 400;
        let doy: u64 = doe - (365 * yoe + yoe / 4 - yoe / 100);
        let mp: u64 = (5 * doy + 2) / 153;
        let d: u64 = doy - (153 * mp + 2) / 5 + 1;
        let m: u64 = if mp < 10 { mp + 3 } else { mp - 9 };
        let y: i64 = if m <= 2 { y + 1 } else { y };
        format!("{:04}-{:02}-{:02}", y, m, d)
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| network → cache file | `nono setup --refresh-trust-root` is the ONLY network call in Phase 32. It fetches TUF metadata from `https://tuf-repo-cdn.sigstore.dev` (sigstore-rs default) and persists the verified `TrustedRoot` to `<nono_home_dir()>/.nono/trust-root/trusted_root.json`. Any compromise of this connection compromises every subsequent verify until the next refresh. |
| cache file → verify | `load_production_trusted_root` reads the cache during every keyless verify. The file lives under `<nono_home_dir()>` (per-user). An attacker with user-level access can swap it; defense-in-depth note in T-32-02-04 below. |
| library API change | `pub async fn` → `pub fn` is a deliberate D-19 deviation enumerated in D-32-15. `tests/integration/test_upstream_drift.sh` is annotated to suppress the drift alarm. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-32-02-01 | S (Spoofing) | TUF cache during refresh | mitigate | sigstore-rs's `TrustedRoot::production()` runs full TUF verification (signature threshold of 3, root-of-trust pinned in upstream). Setup fails fail-closed if any verification step fails. The bytes we persist are post-verification. |
| T-32-02-02 | T (Tampering) | Cached `trusted_root.json` (between refresh and verify) | mitigate | Cache file lives under `<nono_home_dir()>` — per-user permissions apply. An attacker with user-level access can already MITM verify itself; the cache is no worse a target. We DO NOT extend protection to "machine-wide cache writable by user" — that's the explicit per-user trade-off in D-32-01. |
| T-32-02-03 | I (Information disclosure) | Network call from setup | accept | `nono setup --refresh-trust-root` is opt-in (user explicitly invokes). It hits a public endpoint (`tuf-repo-cdn.sigstore.dev`); no PII leaked. The verify path stays offline (D-32-03 invariant tested by `verify_makes_no_outbound_http_calls`). |
| T-32-02-04 | T (Tampering) | First-run / expired-cache fail-closed paths | mitigate | D-32-03 + D-32-05 enforce that absent or expired cache → fail-closed with an explicit recovery message. Permissive defaults (e.g., "fall back to bundled root if cache missing") would re-introduce the original failure mode the phase exists to fix. Tested by `missing_cache_fails_closed` and `expired_cache_fails_closed_with_recovery_hint`. |
| T-32-02-05 | E (Elevation of privilege) | `nono setup --refresh-trust-root` admin check | mitigate | Per D-32-01 + Pitfall 7, the new step does NOT call `is_admin_process()`. Cache is per-user. An attacker tricking a user into running with elevated privileges wouldn't gain anything from the refresh path that they couldn't gain from any other elevated invocation. |
| T-32-02-06 | D (Denial of service) | Verify hangs when TUF down | mitigate | RESEARCH § Anti-Patterns: "Don't put TUF refresh networking in verify path." The rewrite makes verify pure file-I/O. TUF outages affect refresh (a setup-time concern with a clear error path) but never verify. |
| T-32-02-07 | T (Tampering) | Lossy date-prefix string comparison for expiry | accept | RESEARCH Assumption A3: string-prefix comparison `"2026-05-09" < &valid_for_end[..10]` is fail-closed-correct for "is the end-date in the past?" questions. False negatives (saying a current date is expired) cause user-visible refresh prompts — annoying, not insecure. False positives (saying expired is current) require the comparison to break for `YYYY-MM-DD` ASCII-sorted strings, which it doesn't. |
