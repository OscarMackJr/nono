        let user_identity_pattern = args.identity.as_deref().ok_or_else(|| {
            "keyless bundle requires --identity <REGEX> (matched against bundle SAN)".to_string()
        })?;

        // D-32-08 defense-in-depth: URL-component issuer comparison via the
        // existing validate_oidc_issuer (signing.rs:86-130). This rejects
        // prefix attacks like `https://gitlab.com.evil` matching `https://gitlab.com`.
        trust::signing::validate_oidc_issuer(&san_issuer, user_issuer).map_err(|e| {
            format!("OIDC issuer validation failed: {e}")

        // sigstore-verify exact-equality on issuer; identity stays unset
        // because sigstore-verify cannot regex-match (Pitfall 1).

            .with_issuer(user_issuer.to_string());
        trust::verify_bundle_with_digest(
            &file_digest_hex, &bundle, &trusted_root, &sigstore_policy, file_path,
        )
        .map_err(|e| format!("Sigstore verification failed: {e}"))?;
