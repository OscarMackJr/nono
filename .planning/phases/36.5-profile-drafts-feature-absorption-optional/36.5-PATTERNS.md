pub fn write_lockfile(lockfile: &Lockfile) -> Result<()> {
    let path = lockfile_path()?;
    if let Some(parent) = path.parent() {
        fs::create_dir_all(parent).map_err(NonoError::Io)?;
    }
    let tmp_path = path.with_extension("json.tmp");
    let json = serde_json::to_string_pretty(lockfile)
        .map_err(|e| NonoError::ConfigParse(format!("failed to serialize lockfile: {e}")))?;
    fs::write(&tmp_path, format!("{json}\n")).map_err(NonoError::Io)?;
    fs::rename(&tmp_path, &path).map_err(NonoError::Io)?;
    Ok(())
}

```rust
fn cmd_promote(args: ProfilePromoteArgs) -> Result<()> {
    // 1. Validate name (existing helper)
    if !profile::is_valid_profile_name(&args.name) {
        return Err(NonoError::ProfileParse(format!("Invalid profile name '{}'", args.name)));
    }
    // 2. Resolve draft + canonical paths
    let draft_path = profile::get_user_profile_draft_path(&args.name)?;
    let canonical_path = profile::get_user_profile_path(&args.name)?;
    // 3. Shadow-detection (D-36.5-D1) via reserved_profile_source wrapper
    if let Some(source) = reserved_profile_source(&args.name)? {
        return Err(NonoError::ActionRequired {
            expected: canonical_path.display().to_string(),
            actual: draft_path.display().to_string(),
            resolve_via: format!(/* shadow-specific multi-line text per D-36.5-D3 */),
        });
    }
    // 4. Base-hash verification (skip if no canonical — first-time promote)
    if canonical_path.exists() {
        let canonical_bytes = read_regular_file(&canonical_path, "canonical profile")?;
        let actual_hash = sha256_hex(&canonical_bytes);
        let base_path = profile::get_user_profile_draft_base_path(&args.name)?;
        let expected_hash = fs::read_to_string(&base_path)
            .map_err(|e| NonoError::ProfileRead { path: base_path.clone(), source: e })?
            .trim().to_string();
        if !actual_hash.eq_ignore_ascii_case(&expected_hash) {
            return Err(NonoError::ActionRequired {
                expected: expected_hash,
                actual: actual_hash,
                resolve_via: format!("nono profile init --draft --refresh {}", args.name),
            });
        }
        // 5. Render diff via cmd_diff helper (D-36.5-D4)
        // [load both profiles, print diff using cmd_diff's body]
        // 6. Prompt [y/N] unless --yes
        if !args.yes && !confirm("Promote draft?", false)? {
            eprintln!("{}: promote aborted", prefix());
            return Ok(());
        }
    }
    // 7. Atomic rename
    fs::rename(&draft_path, &canonical_path).map_err(NonoError::Io)?;
    println!("{}: promoted to {}", prefix(), canonical_path.display());
    Ok(())
}
```

fn reserved_profile_source(name: &str) -> Result<Option<&'static str>> {
    let pol = policy::load_embedded_policy()?;
    if pol.profiles.contains_key(name) && !profile::is_user_override(name) {
        return Ok(Some("built-in"));
    }
    if profile::get_package_for_profile(name).is_some() {
        return Ok(Some("package-managed"));
    }
    Ok(None)
}

pub fn fetch_package_status(

    package_ref: &PackageRef,
    installed: Option<&str>,
) -> Result<PackageStatusResponse> {
    let mut url = format!(
        "{}/api/v1/packages/{}/{}/status",
        self.base_url,

}

### Symlink-Safe File Read (Pattern 3)

**Source:** Adapted from upstream `829c341a` `profile_cmd.rs::read_regular_file` + fork's `nix` Unix-only cfg gate.
**Apply to:** Commit 2 — `cmd_promote` reads canonical bytes + draft base-hash sidecar.
```rust
fn read_regular_file(path: &Path, label: &str) -> Result<Vec<u8>> {
    let meta = fs::symlink_metadata(path)
        .map_err(|e| NonoError::ProfileRead { path: path.to_path_buf(), source: e })?;
    if meta.file_type().is_symlink() {
        return Err(NonoError::ProfileParse(format!(
            "{label} must not be a symlink: {}", path.display()
        )));
    }
    if !meta.file_type().is_file() {
        return Err(NonoError::ProfileParse(format!(
            "{label} is not a regular file: {}", path.display()
        )));
    }
    let mut options = fs::OpenOptions::new();
    options.read(true);
    #[cfg(unix)]
    {
        use std::os::unix::fs::OpenOptionsExt;
        options.custom_flags(nix::libc::O_NOFOLLOW);
    }
    let mut file = options.open(path)
        .map_err(|e| NonoError::ProfileRead { path: path.to_path_buf(), source: e })?;
    let mut bytes = Vec::new();
    use std::io::Read;
    file.read_to_end(&mut bytes)
        .map_err(|e| NonoError::ProfileRead { path: path.to_path_buf(), source: e })?;
    Ok(bytes)
}
```
