// Source constant (mirror with nono-specific IDs):
const EVENT_LOG_SOURCE: &str = "nono";  // Phase-82-registered Application source
const EVENT_ID_PATH_DENY: u32       = 10001;
const EVENT_ID_NETWORK_DENY: u32    = 10002;
const EVENT_ID_LABEL_VIOLATION: u32 = 10003;
const EVENT_ID_HOOK_FAIL_CLOSED: u32 = 10004;
const EVENT_ID_TELEMETRY_DEGRADED: u32 = 10005;

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
enum EventLogLevel { Information, Warning }

#[cfg(target_os = "windows")]
fn write_security_event_log(level: EventLogLevel, event_id: u32, body: &str) {
    use windows_sys::Win32::System::EventLog::{
        DeregisterEventSource, RegisterEventSourceW, ReportEventW,
        EVENTLOG_INFORMATION_TYPE, EVENTLOG_WARNING_TYPE,
    };
    let source_wide: Vec<u16> = EVENT_LOG_SOURCE
        .encode_utf16().chain(std::iter::once(0u16)).collect();
    // SAFETY: source_wide is null-terminated UTF-16; handle closed before fn returns.
    let handle = unsafe { RegisterEventSourceW(std::ptr::null(), source_wide.as_ptr()) };
    if handle.is_null() {
        eprintln!("nono: telemetry: RegisterEventSourceW returned NULL (source not registered) — {body}");
        return;   // D-03: loud non-fatal, never silent drop
    }
    let body_wide: Vec<u16> = body.encode_utf16().chain(std::iter::once(0u16)).collect();
    let strings: [*const u16; 1] = [body_wide.as_ptr()];
    let event_type = match level {
        EventLogLevel::Warning => EVENTLOG_WARNING_TYPE,
        EventLogLevel::Information => EVENTLOG_INFORMATION_TYPE,
    };
    // SAFETY: handle is valid, strings[0] is null-terminated UTF-16, user-data ptr is null.
    unsafe {
        let _ = ReportEventW(handle, event_type, 0, event_id,
            std::ptr::null_mut(), 1, 0, strings.as_ptr(), std::ptr::null_mut());
        let _ = DeregisterEventSource(handle);
    }
}

#[cfg(not(target_os = "windows"))]
fn write_security_event_log(_level: EventLogLevel, _event_id: u32, _body: &str) {}

use hmac::{Hmac, Mac};
use sha2::Sha256;
type HmacSha256 = Hmac<Sha256>;
// advance_chain replacement:
fn advance_chain(key: &[u8], prev_head: &[u8; 32], event_bytes: &[u8]) -> [u8; 32] {
    let mut mac = HmacSha256::new_from_slice(key)
        .unwrap_or_else(|_| HmacSha256::new_from_slice(&[0u8; 32]).expect("32-byte key always valid"));
    mac.update(TELEMETRY_CHAIN_DOMAIN);
    mac.update(prev_head);
    mac.update(TELEMETRY_EVENT_DOMAIN);
    mac.update(event_bytes);
    let result = mac.finalize().into_bytes();
    let mut out = [0u8; 32];
    out.copy_from_slice(&result);
    out
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Win32 FFI → RegisterEventSourceW | Unsafe raw pointer use; handle must be closed before function returns |
| Event body → ReportEventW | Single insertion string; must contain JSON only, never raw paths (D-02) |
| tracing-etw ETW provider | Self-describing provider — no manifest; admin can logman-query but cannot forge events |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-84-07 | Tampering | ReportEventW insertion string — JSON body | mitigate | D-02: body = serde_json::to_string(SecurityEvent) which already has path_hash not raw path; serialize error → fallback string, never raw fields |
| T-84-08 | Denial of Service | NULL RegisterEventSourceW → blocking behavior | mitigate | D-03: NULL handle → eprintln! + immediate return; no retry, no blocking, no panic |
| T-84-09 | Denial of Service | advance_chain HMAC key length error | mitigate | InvalidLength error → stderr warn + zeroed-key fallback; chain continues with degraded key; confinement unaffected |
| T-84-10 | Information Disclosure | ETW event via tracing::warn! in emit_security_event | mitigate | The inner warn! carries only SecurityEvent fields (already scrubbed/hashed); same redaction guarantees as the Application-log path |
| T-84-11 | Tampering | tracing-etw LayerBuilder ETW provider GUID | accept | GUID auto-derived from "nono" provider name; not secret; attacker can subscribe but cannot inject without provider registration; ETW is monitoring, not enforcement |
| T-84-SC | Tampering | Cargo dep adds (hmac/tracing-etw/eventlog) | mitigate | Operator-approved in Plan 01 Task 1 checkpoint; confirmed crates.io provenance before Cargo.toml edits |
